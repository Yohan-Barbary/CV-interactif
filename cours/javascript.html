JavaScript - Les Fondamentaux

Introduction à JavaScript

JavaScript est un langage de programmation qui permet d'ajouter de l'interactivité aux pages web.

Caractéristiques :

  • Langage interprété
  • Orienté objet et fonctionnel
  • Exécuté côté client
  • Dynamiquement typé

Intégration du JavaScript

Variables et Types de données

Déclaration de variables :

  • let : variable modifiable
  • const : constante
  • var : ancienne déclaration (à éviter)

// Types primitifs
let nombre = 42;              // Number
let texte = "Hello";          // String
let estVrai = true;          // Boolean
let indefini = undefined;     // Undefined
let nul = null;              // Null
let symbole = Symbol('id');   // Symbol

// Types complexes
let tableau = [1, 2, 3];     // Array
let objet = {                // Object
    nom: "John",
    age: 30
};

// Template literals
let nom = "Alice";
let message = `Bonjour ${nom}!`;

// Type coercion
console.log("5" + 3);        // "53"
console.log("5" - 3);        // 2
console.log(Boolean(""));     // false

Conditions et Boucles


// If...else
if (condition) {
    // code
} else if (autreCondition) {
    // code
} else {
    // code
}

// Switch
switch (valeur) {
    case 1:
        // code
        break;
    default:
        // code
}

// Opérateur ternaire
let resultat = condition ? "vrai" : "faux";

// Boucles
for (let i = 0; i < 5; i++) {
    console.log(i);
}

let tableau = [1, 2, 3];
for (let element of tableau) {
    console.log(element);
}

let objet = { a: 1, b: 2 };
for (let key in objet) {
    console.log(key, objet[key]);
}

while (condition) {
    // code
}

do {
    // code
} while (condition);

Fonctions

Types de fonctions :

  • Déclaration de fonction
  • Expression de fonction
  • Fonction fléchée
  • Méthodes

// Déclaration de fonction
function direBonjour(nom) {
    return `Bonjour ${nom}!`;
}

// Expression de fonction
const addition = function(a, b) {
    return a + b;
};

// Fonction fléchée
const multiplication = (a, b) => a * b;

// Paramètres par défaut
function config(options = {}) {
    const { couleur = "noir", taille = "medium" } = options;
    return `Couleur: ${couleur}, Taille: ${taille}`;
}

// Rest parameters
function somme(...nombres) {
    return nombres.reduce((acc, val) => acc + val, 0);
}

// Closure
function compteur() {
    let count = 0;
    return () => ++count;
}
const incrementer = compteur();

Objets et Arrays

Manipulation des objets et tableaux :

  • Création et modification
  • Méthodes d'array
  • Destructuring
  • Spread operator

// Objets
const personne = {
    nom: "Dupont",
    age: 30,
    adresse: {
        rue: "123 rue principale",
        ville: "Paris"
    }
};

// Destructuring
const { nom, age } = personne;
const { adresse: { ville } } = personne;

// Spread operator
const personneComplete = {
    ...personne,
    email: "dupont@email.com"
};

// Arrays et méthodes
const nombres = [1, 2, 3, 4, 5];

// map, filter, reduce
const doubles = nombres.map(n => n * 2);
const pairs = nombres.filter(n => n % 2 === 0);
const somme = nombres.reduce((acc, val) => acc + val, 0);

// Autres méthodes utiles
nombres.push(6);           // Ajoute à la fin
nombres.pop();            // Retire le dernier
nombres.unshift(0);       // Ajoute au début
nombres.shift();          // Retire le premier
nombres.includes(3);      // Vérifie la présence
nombres.find(n => n > 3); // Trouve le premier élément
nombres.some(n => n > 3); // Vérifie si au moins un élément
nombres.every(n => n > 0); // Vérifie si tous les éléments

Événements

Gestion des événements :

  • addEventListener
  • removeEventListener
  • Event bubbling
  • Event delegation

// Ajout d'événements
element.addEventListener('click', function(event) {
    console.log('Cliqué!');
    event.preventDefault(); // Empêche le comportement par défaut
});

// Fonction nommée pour pouvoir la retirer
function handleClick(event) {
    console.log('Cliqué!');
}
element.addEventListener('click', handleClick);
element.removeEventListener('click', handleClick);

// Event bubbling
parent.addEventListener('click', e => {
    console.log('Parent cliqué');
});
child.addEventListener('click', e => {
    e.stopPropagation(); // Arrête la propagation
    console.log('Enfant cliqué');
});

// Event delegation
document.addEventListener('click', e => {
    if (e.target.matches('.btn')) {
        console.log('Bouton cliqué');
    }
});

// Types d'événements courants
element.addEventListener('mouseover', () => {});
element.addEventListener('mouseout', () => {});
element.addEventListener('keydown', () => {});
element.addEventListener('submit', () => {});
element.addEventListener('change', () => {});

Programmation Asynchrone

Gestion de l'asynchrone :

  • Callbacks
  • Promises
  • Async/Await
  • Fetch API

// Callbacks
function chargerDonnees(callback) {
    setTimeout(() => {
        callback('Données chargées');
    }, 1000);
}

// Promises
const maPromesse = new Promise((resolve, reject) => {
    if (condition) {
        resolve('Succès');
    } else {
        reject('Erreur');
    }
});

maPromesse
    .then(resultat => console.log(resultat))
    .catch(erreur => console.error(erreur));

// Async/Await
async function chargerUtilisateur() {
    try {
        const reponse = await fetch('https://api.exemple.com/user');
        const donnees = await reponse.json();
        return donnees;
    } catch (erreur) {
        console.error('Erreur:', erreur);
    }
}

// Fetch API
fetch('https://api.exemple.com/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Erreur:', error));

// Exemple avec async/await et fetch
async function chargerPlusieursRessources() {
    try {
        const [users, posts] = await Promise.all([
            fetch('/api/users').then(r => r.json()),
            fetch('/api/posts').then(r => r.json())
        ]);
        console.log(users, posts);
    } catch (error) {
        console.error('Erreur:', error);
    }
}
